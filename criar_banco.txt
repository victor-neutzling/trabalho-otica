CREATE DATABASE otica;

use otica;

CREATE TABLE produto (ProdutoID int AUTO_INCREMENT, Nome varchar(255), Descricao varchar(255), Preco decimal(7,2), PRIMARY KEY (ProdutoID));

CREATE TABLE cliente (ClienteID int AUTO_INCREMENT, Nome varchar(255), Cpf varchar(11), Email varchar(100), Telefone varchar(20), PRIMARY KEY(ClienteID));

CREATE TABLE pedido (PedidoID int AUTO_INCREMENT, ClienteID int, ValorTotal decimal(7,2), metodo_pagamento varchar(30), ProdutoID int, PRIMARY KEY(PedidoID), FOREIGN KEY(ClienteID) REFERENCES cliente(ClienteID), FOREIGN KEY(ProdutoID) REFERENCES produto(ProdutoID));

CREATE TABLE funcionario (FuncionarioID int AUTO_INCREMENT primary key, nome varchar (100), cpf varchar(20), cargo varchar (50), salario decimal(10,2));
